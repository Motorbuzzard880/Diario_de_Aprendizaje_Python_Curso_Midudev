# continue, que lo hace es saltar esa iteración en concreto
# y continuar con el bucle

contador = 0
while contador < 10:
    contador += 1

    if contador % 2 == 0:
        continue

    print(contador)


contador = 0
while contador < 5:
    print(contador)
    contador += 1
else:
    print("El bucle ha terminado")
